#JSON JavaScript object Notation

##基本语法
###定义与访问、更改
var json={
    key: value,
    key: value,
    key:{
        key: value,
        key: value,
        key: value
    },
    key:value
};
//两种写法可以交叉使用
console.log(json.key);
console.log(json["key"]);

console.log(json.key.key);

###添加、删除
json.newkey=value;
delete json.key;

###for...in 遍历
	for(var k in obj){
	 	console.log(k + "的值是" + obj[k]);
	 }
